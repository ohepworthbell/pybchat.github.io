let readable_date = (d) => {
  let _date = new Date(Date.parse(d));
  return _date.toDateString()
};

function Meeting(location, d) {
  document.write(`The next meeting is on ${readable_date(d)} at ${location}`);
}

let next_meeting = new Meeting('TBA', '1970-01-01');
>> The next meeting is on Thu Jan 01 1970 at TBA
